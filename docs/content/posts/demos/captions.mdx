import Playground from '../../../components/Playground.astro';
import PlaygroundButton from '../../../components/PlaygroundButton.astro';
import PlaygroundCheckbox from '../../../components/PlaygroundCheckbox.astro';

import { IFRAME_URL } from '../../../variables';

<Playground url={IFRAME_URL}>
  <div slot="actions">
    <div className="playground-actions-container">
      <PlaygroundButton type="get" command="getClosedCaptions">Get caption tracks</PlaygroundButton>
      <PlaygroundButton type="get" command="getClosedCaptionsLanguage">Get current language</PlaygroundButton>
      <PlaygroundButton type="set" command="setClosedCaptionsLanguage" value="">Remove CC</PlaygroundButton>
      <PlaygroundButton type="set" command="setClosedCaptionsLanguage" value="en">Set English</PlaygroundButton>
      <PlaygroundButton type="set" command="setClosedCaptionsLanguage" value="fr">Set French</PlaygroundButton>
      <PlaygroundButton type="set" command="setClosedCaptionsLanguage" value="it">Set Italian</PlaygroundButton>
      <PlaygroundButton type="set" command="setClosedCaptionsLanguage" value="es">Set Spanish</PlaygroundButton>
    </div>
    <hr/>
    <PlaygroundCheckbox id="addEventListener" checked={true} event="closedcaptionslanguagechange">Listen for `closedcaptionslanguagechange` events</PlaygroundCheckbox>
  </div>
  <div slot="code">
    ```js
    import { PlayerSdk } from '@qumu/player-sdk';

    const iframe = document.querySelector('iframe');
    const sdk = new PlayerSdk(iframe);

    document.getElementById('getCaptions').addEventListener('click', () => {
      sdk.getClosedCaptions()
        .then((tracks) => {
          console.log('captions', tracks)
        });
    });

    document.getElementById('getCaptionsLanguage').addEventListener('click', () => {
      sdk.getClosedCaptionsLanguage()
        .then((language) => {
          console.log('captions language', language)
        });
    });

    document.getElementById('setCaptionsNone').addEventListener('click', () => {
      sdk.setClosedCaptionsLanguage('');
    });

    document.getElementById('setEn').addEventListener('click', () => {
      sdk.setClosedCaptionsLanguage('en');
    });

    document.getElementById('setFr').addEventListener('click', () => {
      sdk.setClosedCaptionsLanguage('fr');
    });

    document.getElementById('setIt').addEventListener('click', () => {
      sdk.setClosedCaptionsLanguage('it');
    });

    document.getElementById('setEs').addEventListener('click', () => {
      sdk.setClosedCaptionsLanguage('es');
    });

    sdk.addEventListener('closedcaptionslanguagechange', (newLanguage) => {
      console.log('closedcaptionslanguagechange', newLanguage);
    });
    ```
  </div>
</Playground>
