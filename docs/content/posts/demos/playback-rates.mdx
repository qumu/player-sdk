import Playground from '../../../components/Playground.astro';
import PlaygroundButton from '../../../components/PlaygroundButton.astro';
import PlaygroundCheckbox from '../../../components/PlaygroundCheckbox.astro';
import { IFRAME_URL } from '../../../variables';

<Playground url={IFRAME_URL}>
  <div slot="actions">
    <div className="playground-actions-container">
      <PlaygroundButton type="get" command="getPlaybackRates">Get playback rates</PlaygroundButton>
      <PlaygroundButton type="get" command="getPlaybackRate">Get current playback rate</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={0.25}>0.25x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={0.5}>0.5x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={0.75}>0.75x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={1}>1x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={1.25}>1.25x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={1.5}>1.5x</PlaygroundButton>
      <PlaygroundButton type="set" command="setPlaybackRate" value={2}>2x</PlaygroundButton>
    </div>
    <hr/>
    <PlaygroundCheckbox id="addEventListener" checked={true} event="playbackratechange">Listen for `playbackratechange` events</PlaygroundCheckbox>
  </div>
  <div slot="code">
    ```js
    import { PlayerSdk } from '@qumu/player-sdk';

    const iframe = document.querySelector('iframe');
    const sdk = new PlayerSdk(iframe);

    document.getElementById('getPlaybackRates').addEventListener('click', () => {
      sdk.getPlaybackRates()
        .then((playbackRates) => {
          console.log('Playback rates', playbackRates)
        });
    });

    document.getElementById('getPlaybackRate').addEventListener('click', () => {
      sdk.getPlaybackRate()
        .then((playbackRate) => {
          console.log('Current playback rate', playbackRate)
        });
    });

    document.getElementById('setPlaybackRate25').addEventListener('click', () => {
      sdk.setPlaybackRate(0.25);
    });

    document.getElementById('setPlaybackRate50').addEventListener('click', () => {
      sdk.setPlaybackRate(0.5);
    });

    document.getElementById('setPlaybackRate75').addEventListener('click', () => {
      sdk.setPlaybackRate(0.75);
    });

    document.getElementById('setPlaybackRate100').addEventListener('click', () => {
      sdk.setPlaybackRate(1);
    });

    document.getElementById('setPlaybackRate125').addEventListener('click', () => {
      sdk.setPlaybackRate(1.25);
    });

    document.getElementById('setPlaybackRate150').addEventListener('click', () => {
      sdk.setPlaybackRate(1.5);
    });

    document.getElementById('setPlaybackRate200').addEventListener('click', () => {
      sdk.setPlaybackRate(2);
    });

    sdk.addEventListener('playbackratechange', (newPlaybackRate) => {
      console.log('playbackratechange', newPlaybackRate);
    });
    ```
  </div>
</Playground>
