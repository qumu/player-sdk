import Playground from '../../../components/Playground.astro';
import PlaygroundButton from '../../../components/PlaygroundButton.astro';
import PlaygroundCheckbox from '../../../components/PlaygroundCheckbox.astro';
import { IFRAME_URL } from '../../../variables';

<Playground url={IFRAME_URL}>
  <div slot="actions" className="actions">
    <PlaygroundButton type="get" command="getVolume">Get volume</PlaygroundButton>
    <PlaygroundButton type="set" command="setVolume" value={0}>Set volume to 0</PlaygroundButton>
    <PlaygroundButton type="set" command="setVolume" value={50}>Set volume to 50%</PlaygroundButton>
    <PlaygroundButton type="set" command="setVolume" value={100}>Set volume to 100%</PlaygroundButton>
    <hr/>
    <PlaygroundCheckbox id="addEventListener" checked={true} event="volumechange">Listen for `volumechange` events</PlaygroundCheckbox>
  </div>
  <div slot="code">
    ```js
    import { PlayerSdk } from '@qumu/player-sdk';

    const iframe = document.querySelector('iframe');
    const sdk = new PlayerSdk(iframe);

    document.getElementById('getVolume').addEventListener('click', () => {
      sdk.getVolume()
        .then((volume) => {
          console.log('volume', volume)
        });
    });

    document.getElementById('setVolume0').addEventListener('click', () => {
      sdk.setVolume(0);
    });

    document.getElementById('setVolume50').addEventListener('click', () => {
      sdk.setVolume(50);
    });

    document.getElementById('setVolume100').addEventListener('click', () => {
      sdk.setVolume(100);
    });

    sdk.addEventListener('volumechange', (newVolume) => {
      console.log('volumechange', newVolume);
    });
    ```
  </div>
</Playground>
