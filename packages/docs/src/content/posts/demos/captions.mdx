import Playground from '../../../components/Playground.astro';
import PlaygroundButton from '../../../components/PlaygroundButton.astro';
import PlaygroundCheckbox from '../../../components/PlaygroundCheckbox.astro';

import { IFRAME_URL } from '../../../variables';

<Playground url={IFRAME_URL}>
  <div slot="actions">
    <div className="playground-actions-container">
      <PlaygroundButton type="get" command="getCaptionTracks">Get caption tracks</PlaygroundButton>
      <PlaygroundButton type="get" command="getCurrentCaptionTrack">Get current caption track</PlaygroundButton>
      <PlaygroundButton type="set" command="disableCaptionTrack" value={null}>Remove CC</PlaygroundButton>
      <PlaygroundButton type="set" command="enableCaptionTrack" value="en">Set English</PlaygroundButton>
      <PlaygroundButton type="set" command="enableCaptionTrack" value="fr">Set French</PlaygroundButton>
      <PlaygroundButton type="set" command="enableCaptionTrack" value="it">Set Italian</PlaygroundButton>
      <PlaygroundButton type="set" command="enableCaptionTrack" value="es">Set Spanish</PlaygroundButton>
    </div>
    <hr/>
    <PlaygroundCheckbox id="addEventListener" checked={true} event="captiontrackchange">Listen for `captiontrackchange` events</PlaygroundCheckbox>
  </div>
  <div slot="code">
    ```js
    import { PlayerSdk } from '@enghouse-qumu/player-sdk';

    const iframe = document.querySelector('iframe');
    const sdk = new PlayerSdk(iframe);

    document.getElementById('getCaptionTracks').addEventListener('click', () => {
      sdk.getCaptionTracks()
        .then((captionTracks) => {
          console.log('caption tracks', captionTracks)
        });
    });

    document.getElementById('getCurrentCaptionTrack').addEventListener('click', () => {
      sdk.getCurrentCaptionTrack()
        .then((captionTrack) => {
          console.log('current captionTrack', captionTrack)
        });
    });

    document.getElementById('disableTextTracks').addEventListener('click', () => {
      sdk.disableTextTracks();
    });

    document.getElementById('setEn').addEventListener('click', () => {
      sdk.enableCaptionTrack('en');
    });

    document.getElementById('setFr').addEventListener('click', () => {
      sdk.enableCaptionTrack('fr');
    });

    document.getElementById('setIt').addEventListener('click', () => {
      sdk.enableCaptionTrack('it');
    });

    document.getElementById('setEs').addEventListener('click', () => {
      sdk.enableCaptionTrack('es');
    });

    sdk.addEventListener('captiontrackchange', (newLanguage) => {
      console.log('captiontrackchange', newLanguage);
    });
    ```
  </div>
</Playground>
